CREATE DATABASE barn;

CREATE TABLE rooster (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    dateOfBirth TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    dateOfDeath TIMESTAMP DEFAULT CURRENT_TIMESTAMP, 
);

CREATE TABLE hen (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    dateOfBirth TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    dateOfDeath TIMESTAMP DEFAULT CURRENT_TIMESTAMP, 
);

CREATE TABLE chick (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    dateOfBirth TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    dateOfDeath TIMESTAMP DEFAULT CURRENT_TIMESTAMP, 
);
